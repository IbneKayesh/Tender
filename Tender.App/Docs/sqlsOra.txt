    SUPPLIER INTEGER DEFAULT 1, 
    PURCHASER INTEGER DEFAULT 0, 
    SUPPLIER_NOTIFY INTEGER DEFAULT 1, 
    PURCHASER_NOTIFY INTEGER DEFAULT 0,    
    IS_CONFIRMED INTEGER DEFAULT 0,
    VENDOR_USER_ID NVARCHAR2(50) NOT NULL UNIQUE,
    CONTACT_NAME NVARCHAR2(50) , 
    CONTACT_NUMBER NVARCHAR2(50) ,
    ADDRESS_1 NVARCHAR2(100) ,
    ADDRESS_2 NVARCHAR2(100) ,
    YEAR_OF_ESTABLISHMENT NUMBER ,
    YEARLY_TRUNOVER NUMBER ,
    NO_OF_EMPLOYEES NVARCHAR2(70) , 
    TAX_NUMBER NVARCHAR2(50) ,
    TRADE_NUMBER NVARCHAR2(50) ,
    IS_APPROVE INTEGER                     DEFAULT 0 
)
TABLESPACE DATA01;
